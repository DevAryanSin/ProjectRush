        /** ==========================================
         * SOUND ENGINE
         * ========================================== */
        const SoundEngine = {
            ctx: null,
            init() { if (this.ctx) return; this.ctx = new (window.AudioContext || window.webkitAudioContext)(); },
            playShot() {
                this.init(); const now = this.ctx.currentTime;
                const bufferSize = this.ctx.sampleRate * 0.15;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
                const noise = this.ctx.createBufferSource(); noise.buffer = buffer;
                const noiseGain = this.ctx.createGain();
                noiseGain.gain.setValueAtTime(0.5, now); noiseGain.gain.exponentialRampToValueAtTime(0.001, now + 0.12);
                noise.connect(noiseGain); noiseGain.connect(this.ctx.destination); noise.start();
                const osc = this.ctx.createOscillator(); const oscGain = this.ctx.createGain();
                osc.type = 'triangle'; osc.frequency.setValueAtTime(100, now); osc.frequency.exponentialRampToValueAtTime(30, now + 0.1);
                oscGain.gain.setValueAtTime(0.6, now); oscGain.gain.exponentialRampToValueAtTime(0.001, now + 0.12);
                osc.connect(oscGain); oscGain.connect(this.ctx.destination); osc.start(); osc.stop(now + 0.12);
            },
            playHit() {
                this.init(); const now = this.ctx.currentTime;
                const osc = this.ctx.createOscillator(); const gain = this.ctx.createGain();
                osc.type = 'sine'; osc.frequency.setValueAtTime(800, now);
                gain.gain.setValueAtTime(0.3, now); gain.gain.exponentialRampToValueAtTime(0.001, now + 0.1);
                osc.connect(gain); gain.connect(this.ctx.destination); osc.start(); osc.stop(now + 0.1);
            },
            playClick() {
                this.init(); const now = this.ctx.currentTime;
                const osc = this.ctx.createOscillator(); const gain = this.ctx.createGain();
                osc.type = 'square'; osc.frequency.setValueAtTime(1200, now);
                gain.gain.setValueAtTime(0.05, now); gain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
                osc.connect(gain); gain.connect(this.ctx.destination); osc.start(); osc.stop(now + 0.04);
            },
            playHiss() {
                this.init(); const now = this.ctx.currentTime;
                const bufferSize = this.ctx.sampleRate * 0.5;
                const buffer = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
                const data = buffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
                const noise = this.ctx.createBufferSource(); noise.buffer = buffer;
                const filter = this.ctx.createBiquadFilter();
                filter.type = 'bandpass'; filter.frequency.setValueAtTime(4000, now); filter.Q.setValueAtTime(1, now);
                const noiseGain = this.ctx.createGain();
                noiseGain.gain.setValueAtTime(0, now);
                noiseGain.gain.linearRampToValueAtTime(0.15, now + 0.1);
                noiseGain.gain.linearRampToValueAtTime(0, now + 0.5);
                noise.connect(filter); filter.connect(noiseGain); noiseGain.connect(this.ctx.destination);
                noise.start();
            },
            playHurt() {
                this.init(); const now = this.ctx.currentTime;
                const osc = this.ctx.createOscillator(); const gain = this.ctx.createGain();
                osc.type = 'sawtooth'; osc.frequency.setValueAtTime(60, now);
                gain.gain.setValueAtTime(0.4, now); gain.gain.exponentialRampToValueAtTime(0.001, now + 0.4);
                osc.connect(gain); gain.connect(this.ctx.destination); osc.start(); osc.stop(now + 0.4);
                this.playHiss(); 
            }
        };

        /** ==========================================
         * GAME LOGIC
         * ========================================== */
        const Game = {
            state: { position: 1, hp: 5, maxHp: 5, bullets: 0, phase: 'IDLE', ladders: {}, snakes: {}, tailFallback: null },
            config: { total: 100, cols: 10, boardSize3D: 60 },

            init() {
                this.generateBoard();
                this.draw2DBoard();
                this.updateUI();
                ThreeEngine.init();
            },

            generateBoard() {
                this.state.ladders = {}; this.state.snakes = {};
                const used = new Set();
                const addPiece = (isLadder, minLen, maxLen, forcedRange = null) => {
                    let head, tail, len;
                    let attempts = 0;
                    while (attempts < 100) {
                        if (forcedRange) head = Math.floor(Math.random() * (forcedRange.hMax - forcedRange.hMin + 1)) + forcedRange.hMin;
                        else head = Math.floor(Math.random() * 90) + 10;
                        len = Math.floor(Math.random() * (maxLen - minLen + 1)) + minLen;
                        tail = isLadder ? head + len : head - len;
                        const headRow = Math.floor((head - 1) / 10);
                        const tailRow = Math.floor((tail - 1) / 10);
                        if (tail > 1 && tail < 100 && head > 1 && head < 100 && !used.has(head) && !used.has(tail) && headRow !== tailRow) { 
                            if (isLadder) this.state.ladders[head] = tail;
                            else this.state.snakes[head] = tail;
                            used.add(head); used.add(tail);
                            return true;
                        }
                        attempts++;
                    }
                    return false;
                };
                addPiece(true, 10, 20, { hMin: 2, hMax: 15 });
                for(let i=0; i<2; i++) addPiece(false, 40, 60, { hMin: 70, hMax: 98 });
                for(let i=0; i<3; i++) addPiece(false, 15, 30);
                for(let i=0; i<3; i++) addPiece(false, 11, 15);
                for(let i=0; i<8; i++) addPiece(true, 11, 25);
            },

            getCoords(sq) {
                const i = sq - 1;
                const r = Math.floor(i / this.config.cols);
                let c = i % this.config.cols;
                if (r % 2 !== 0) c = 9 - c;
                const pctX = c * 10 + 5; const pctY = (9 - r) * 10 + 5;
                const bs = this.config.boardSize3D;
                const wX = (pctX / 100) * bs - (bs / 2);
                const wZ = (pctY / 100) * bs - (bs / 2);
                return { pctX, pctY, wX, wZ };
            },

            draw2DBoard() {
                const grid = document.getElementById('grid');
                const svg = document.getElementById('svg-layer');
                grid.innerHTML = ''; svg.innerHTML = '';
                for (let r = 9; r >= 0; r--) {
                    for (let c = 0; c < 10; c++) {
                        const cell = document.createElement('div');
                        const num = (r % 2 === 0) ? (r * 10 + c + 1) : (r * 10 + (9 - c) + 1);
                        const isEven = (r + c) % 2 === 0;
                        cell.className = `cell ${isEven ? 'cell-even' : 'cell-odd'}`;
                        cell.innerText = num;
                        grid.appendChild(cell);
                    }
                }
                for (const [s, e] of Object.entries(this.state.ladders)) {
                    const c1 = this.getCoords(parseInt(s)), c2 = this.getCoords(e);
                    const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');
                    line.setAttribute('x1', `${c1.pctX}%`); line.setAttribute('y1', `${c1.pctY}%`);
                    line.setAttribute('x2', `${c2.pctX}%`); line.setAttribute('y2', `${c2.pctY}%`);
                    line.setAttribute('stroke', '#a0522d'); line.setAttribute('stroke-width', '2.5'); 
                    line.setAttribute('stroke-dasharray', '1,4'); line.setAttribute('class', 'ladder-line');
                    svg.appendChild(line);
                }
                for (const [h, t] of Object.entries(this.state.snakes)) {
                    const c1 = this.getCoords(parseInt(h)), c2 = this.getCoords(t);
                    const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
                    const mx = (c1.pctX+c2.pctX)/2 + (Math.random()*30-15);
                    const my = (c1.pctY+c2.pctY)/2 + (Math.random()*30-15);
                    path.setAttribute('d', `M ${c1.pctX} ${c1.pctY} Q ${mx} ${my} ${c2.pctX} ${c2.pctY}`);
                    path.setAttribute('fill', 'none'); path.setAttribute('stroke', '#1a331a'); 
                    path.setAttribute('stroke-width', '4'); 
                    svg.appendChild(path);
                    const stripes = path.cloneNode();
                    stripes.setAttribute('stroke', '#90ee90');
                    stripes.setAttribute('stroke-width', '2');
                    stripes.setAttribute('stroke-dasharray', '4,4');
                    svg.appendChild(stripes);
                }
                this.updateToken(false);
            },

            updateToken(anim = true) {
                const c = this.getCoords(this.state.position);
                const t = document.getElementById('player-token');
                if (anim) return gsap.to(t, { left: `${c.pctX}%`, top: `${c.pctY}%`, duration: 0.5, ease: "back.out(1.2)" });
                else { t.style.left = `${c.pctX}%`; t.style.top = `${c.pctY}%`; return Promise.resolve(); }
            },

            updateUI() {
                const hpCont = document.getElementById('player-hp-board');
                if (!hpCont) return;
                hpCont.innerHTML = '';
                for(let i=0; i<this.state.maxHp; i++) {
                    const h = document.createElement('span');
                    h.className = `hp-heart ${i >= this.state.hp ? 'empty' : ''}`;
                    h.innerHTML = '♥'; hpCont.appendChild(h);
                }
                document.getElementById('board-ammo').innerText = this.state.bullets;
                const av = document.getElementById('ammo-visual');
                if (av) {
                    av.innerHTML = '';
                    for(let i=0; i<this.state.bullets; i++) {
                        const b = document.createElement('div'); b.className = 'ammo-bullet'; av.appendChild(b);
                    }
                }
            },

            setStatus(msg) { document.getElementById('game-status').innerText = msg; },

            async rollDice() {
                if (this.state.phase !== 'IDLE') return;
                this.state.phase = 'ROLLING'; SoundEngine.playClick();
                const r = Math.floor(Math.random() * 6) + 1;
                this.setStatus("Consulting the fates...");
                const rots = { 1:{x:0,y:0}, 2:{x:0,y:-90}, 3:{x:-90,y:0}, 4:{x:90,y:0}, 5:{x:0,y:90}, 6:{x:0,y:180} };
                document.getElementById('dice').style.transform = `rotateX(${rots[r].x+1080}deg) rotateY(${rots[r].y+1080}deg)`;
                await new Promise(res => setTimeout(res, 1200));
                this.state.bullets = r; this.updateUI();
                await this.movePlayer(r);
            },

            async movePlayer(steps) {
                this.state.phase = 'MOVING';
                for (let i = 0; i < steps; i++) {
                    if (this.state.position === this.config.total) break;
                    this.state.position++; await this.updateToken();
                }
                this.evalSquare();
            },

            async evalSquare() {
                const p = this.state.position;
                if (p === this.config.total) { document.getElementById('win-modal').style.display = 'block'; return; }
                if (this.state.ladders[p]) {
                    this.setStatus("Scaling the structure...");
                    this.state.position = this.state.ladders[p];
                    await new Promise(r => setTimeout(r, 400)); await this.updateToken();
                }
                if (this.state.snakes[p]) {
                    const tail = this.state.snakes[p];
                    this.state.tailFallback = tail;
                    this.setStatus("A guardian appears!");
                    setTimeout(() => ThreeEngine.startCombat(p, tail), 800);
                } else {
                    this.state.phase = 'IDLE'; this.setStatus("Your move, explorer.");
                }
            },

            forceSnake() {
                if (this.state.phase === 'COMBAT') return;
                this.state.phase = 'COMBAT';
                const tail = Math.max(1, this.state.position - 15);
                this.state.tailFallback = tail;
                this.state.bullets = 6; this.updateUI();
                ThreeEngine.startCombat(this.state.position, tail);
            },

            respawn() {
                document.getElementById('death-modal').style.display = 'none';
                this.state.hp = this.state.maxHp; this.state.bullets = 0;
                this.state.position = this.state.tailFallback || 1;
                this.updateToken(true).then(() => { this.state.phase = 'IDLE'; this.updateUI(); });
            },

            reset() { 
                document.getElementById('win-modal').style.display = 'none';
                this.state.position = 1;
                this.state.hp = this.state.maxHp;
                this.state.bullets = 0;
                this.state.phase = 'IDLE';
                this.generateBoard();
                this.draw2DBoard();
                this.updateUI();
                this.updateToken(false);
                this.setStatus("New expedition started!");
            }
        };

         * 3D ENGINE

        const ThreeEngine = {
            scene: null, camera: null, renderer: null, clock: new THREE.Clock(),
            targetGroup: null, gunGroup: null, boardGroup: null, tongue: null,
            crawlingSnakes: [],
            combat: { active: false, enemyHp: 0 },
            mouseSensitivity: 0.0012, 
            isMouseLocked: false,

            init() {
                this.scene = new THREE.Scene();
                this.scene.background = new THREE.Color(0x0a0500); 
                this.camera = new THREE.PerspectiveCamera(70, window.innerWidth / window.innerHeight, 0.1, 1000);
                this.camera.rotation.order = 'YXZ'; 
                this.renderer = new THREE.WebGLRenderer({ antialias: true });
                this.renderer.setSize(window.innerWidth, window.innerHeight);
                const layer = document.getElementById('fps-layer');
                if (layer.children.length === 0) layer.appendChild(this.renderer.domElement);
                
                this.scene.add(new THREE.AmbientLight(0x221a10, 0.4)); 
                const centralLight = new THREE.PointLight(0xff9944, 2.5, 120); 
                centralLight.position.set(0, 20, 0); 
                this.scene.add(centralLight);

                this.buildRevolver();
                this.buildEnvironment();

                const hud = document.getElementById('fps-hud');
                hud.addEventListener('mousedown', (e) => {
                    if (e.button === 0) {
                        if (!this.isMouseLocked) hud.requestPointerLock();
                        else this.handleFire();
                    }
                });

                document.addEventListener('pointerlockchange', () => {
                    this.isMouseLocked = document.pointerLockElement === hud;
                });

                document.addEventListener('mousemove', (e) => {
                    if (this.isMouseLocked && this.combat.active) {
                        const mx = e.movementX || 0;
                        const my = e.movementY || 0;
                        if (Math.abs(mx) > 300 || Math.abs(my) > 300) return;
                        this.camera.rotation.y -= mx * this.mouseSensitivity;
                        this.camera.rotation.x -= my * this.mouseSensitivity;
                        this.camera.rotation.x = Math.max(-Math.PI/2.2, Math.min(Math.PI/2.2, this.camera.rotation.x));
                    }
                });

                window.addEventListener('resize', () => {
                    this.camera.aspect = window.innerWidth / window.innerHeight;
                    this.camera.updateProjectionMatrix();
                    this.renderer.setSize(window.innerWidth, window.innerHeight);
                });
                this.animate();
            },

            buildRuinedWall(zPos) {
                const stoneMat = new THREE.MeshStandardMaterial({ color: 0x2d2d2a, roughness: 0.9 });
                const vineMat = new THREE.MeshStandardMaterial({ color: 0x1a2e1a, roughness: 1 });
                const wallGroup = new THREE.Group();
                const brickW = 8, brickH = 4, brickD = 2;
                
                for(let y = 0; y < 10; y++) {
                    for(let x = -13; x <= 13; x++) {
                        if(Math.random() > 0.85 && (y > 4 || Math.abs(x) > 8)) continue;
                        
                        const brick = new THREE.Mesh(
                            new THREE.BoxGeometry(brickW * 0.95, brickH * 0.95, brickD),
                            stoneMat
                        );
                        brick.position.set(x * brickW + (y % 2 * 2), y * brickH, zPos + (Math.random() * 0.2 - 0.1));
                        brick.rotation.z = Math.random() * 0.04 - 0.02;
                        wallGroup.add(brick);

                        if(Math.random() > 0.96) {
                            const vLen = Math.random() * 15 + 5;
                            const vine = new THREE.Mesh(new THREE.CylinderGeometry(0.1, 0.08, vLen), vineMat);
                            vine.position.set(brick.position.x, brick.position.y - vLen/2, brick.position.z + 1.1);
                            wallGroup.add(vine);
                        }
                    }
                }
                this.scene.add(wallGroup);
            },

            createSmallSnake(parent, x, y, z) {
                const sGroup = new THREE.Group();
                const snakeMat = new THREE.MeshStandardMaterial({ color: 0x1a331a });
                const stripeMat = new THREE.MeshStandardMaterial({ color: 0x90ee90 });
                
                const segments = [];
                for(let i=0; i<6; i++) {
                    // FIX: Replaced CapsuleGeometry with CylinderGeometry for r128 compatibility
                    const seg = new THREE.Mesh(new THREE.CylinderGeometry(0.15, 0.15, 0.4, 8), i % 2 === 0 ? snakeMat : stripeMat);
                    seg.position.z = i * -0.4;
                    seg.rotation.x = Math.PI/2;
                    sGroup.add(seg);
                    segments.push(seg);
                }
                
                sGroup.position.set(x, y + 0.1, z);
                sGroup.rotation.y = Math.random() * Math.PI * 2;
                parent.add(sGroup);
                this.crawlingSnakes.push({ group: sGroup, segments, offset: Math.random() * 100 });
            },

            buildEnvironment() {
                const floorGeo = new THREE.PlaneGeometry(400, 400);
                const floorMat = new THREE.MeshStandardMaterial({ color: 0x1a0f08, roughness: 1 });
                const floor = new THREE.Mesh(floorGeo, floorMat);
                floor.rotation.x = -Math.PI / 2; floor.position.y = -0.5; this.scene.add(floor);
                
                const stoneMat = new THREE.MeshStandardMaterial({ color: 0x2d2d2a, roughness: 0.9 }); 
                const vineMat = new THREE.MeshStandardMaterial({ color: 0x1b2e1b, roughness: 1 }); 

                this.buildRuinedWall(-75);
                this.buildRuinedWall(75);

                for (let side = -1; side <= 1; side += 2) {
                    if (side === 0) continue;
                    for (let tier = 0; tier < 6; tier++) {
                        const stand = new THREE.Mesh(new THREE.BoxGeometry(10, 2.5, 120), stoneMat);
                        stand.position.set(side * (45 + tier * 10), tier * 2 - 1, 0);
                        this.scene.add(stand);

                        const snakeCount = Math.floor(Math.random() * 4) + 2;
                        for(let s=0; s<snakeCount; s++) {
                            this.createSmallSnake(
                                this.scene, 
                                side * (45 + tier * 10) + (Math.random()*8-4),
                                tier * 2 + 0.25,
                                Math.random() * 100 - 50
                            );
                        }
                        
                        if (tier === 5) {
                            for (let z = -50; z <= 50; z += 25) {
                                const pGroup = new THREE.Group();
                                const pillar = new THREE.Mesh(new THREE.BoxGeometry(3, 15, 3), stoneMat);
                                pGroup.add(pillar);
                                const fire = new THREE.Mesh(new THREE.SphereGeometry(1.0), new THREE.MeshBasicMaterial({ color: 0xff6600 }));
                                fire.position.y = 8;
                                pGroup.add(fire);
                                const pLight = new THREE.PointLight(0xff4400, 1.2, 35);
                                pLight.position.copy(fire.position);
                                pGroup.add(pLight);
                                for(let v=0; v<3; v++) {
                                    const vine = new THREE.Mesh(new THREE.CylinderGeometry(0.1, 0.15, 15), vineMat);
                                    vine.rotation.z = Math.random() * 0.2 - 0.1;
                                    vine.position.set(Math.random()-0.5, 0, 1.6);
                                    pGroup.add(vine);
                                }
                                pGroup.position.set(side * (45 + tier * 10), 7.5, z);
                                this.scene.add(pGroup);
                            }
                        }
                    }
                }

                const statuePositions = [[-80, 0, -50], [80, 0, -50], [-80, 0, 50],];
                statuePositions.forEach(pos => {
                    const statue = new THREE.Group();
                    const base = new THREE.Mesh(new THREE.BoxGeometry(8, 12, 8), stoneMat);
                    statue.add(base);
                    const body = new THREE.Mesh(new THREE.CylinderGeometry(3, 4, 15, 8), stoneMat);
                    body.position.y = 13.5;
                    statue.add(body);
                    const head = new THREE.Mesh(new THREE.BoxGeometry(4, 5, 4), stoneMat);
                    head.position.y = 23;
                    statue.add(head);
                    const eL = new THREE.Mesh(new THREE.SphereGeometry(0.3), new THREE.MeshBasicMaterial({color: 0x44ff44}));
                    eL.position.set(-0.8, 23.5, 2.1); statue.add(eL);
                    const eR = eL.clone(); eR.position.x = 0.8; statue.add(eR);
                    statue.position.set(pos, 0, pos); // Fixed pos mapping
                    statue.lookAt(0, 0, 0);
                    this.scene.add(statue);
                });

                for(let i=0; i<25; i++) {
                    const rubble = new THREE.Mesh(new THREE.IcosahedronGeometry(Math.random() * 2 + 1, 0), stoneMat);
                    rubble.position.set(Math.random()*160-80, 0, Math.random()*160-80);
                    rubble.rotation.set(Math.random(), Math.random(), Math.random());
                    this.scene.add(rubble);
                }
            },

            buildBoard() {
                if (this.boardGroup) this.scene.remove(this.boardGroup);
                this.boardGroup = new THREE.Group();
                const bs = Game.config.boardSize3D;
                const cs = bs / 10;
                const slabGeo = new THREE.BoxGeometry(cs * 0.95, 0.4, cs * 0.95);
                for(let i=1; i<=100; i++) {
                    const c = Game.getCoords(i);
                    const mat = new THREE.MeshStandardMaterial({ color: (Math.floor((i-1)/10) + (i-1)%10) % 2 === 0 ? 0x222220 : 0x2a2a28 });
                    const m = new THREE.Mesh(slabGeo, mat); m.position.set(c.wX, -0.2, c.wZ);
                    this.boardGroup.add(m);
                }
                this.scene.add(this.boardGroup);
            },

            buildRevolver() {
                this.gunGroup = new THREE.Group();
                const ironMat = new THREE.MeshStandardMaterial({ color: 0x333333, metalness: 0.9, roughness: 0.2 });
                const woodMat = new THREE.MeshStandardMaterial({ color: 0x3d2b1f, roughness: 0.8 });
                const barrel = new THREE.Mesh(new THREE.CylinderGeometry(0.06, 0.06, 0.8, 8), ironMat);
                barrel.rotation.x = Math.PI/2; barrel.position.z = -0.4; this.gunGroup.add(barrel);
                const cylinder = new THREE.Mesh(new THREE.CylinderGeometry(0.12, 0.12, 0.3, 12), ironMat);
                cylinder.rotation.x = Math.PI/2; this.gunGroup.add(cylinder);
                const grip = new THREE.Mesh(new THREE.BoxGeometry(0.1, 0.4, 0.2), woodMat);
                grip.position.set(0, -0.3, 0.1); grip.rotation.x = Math.PI/6; this.gunGroup.add(grip);
                this.gunGroup.position.set(0.5, -0.5, -0.5);
                this.camera.add(this.gunGroup); this.scene.add(this.camera);
            },

            spawnTarget(square, initial = false) {
                const c = Game.getCoords(square);
                if (this.targetGroup) this.scene.remove(this.targetGroup);
                
                this.targetGroup = new THREE.Group();
                const snakeGreen = new THREE.MeshStandardMaterial({ color: 0x1a331a, roughness: 0.7 });
                const lightGreen = new THREE.MeshStandardMaterial({ color: 0x90ee90 });
                const glowingRed = new THREE.MeshBasicMaterial({ color: 0xff0000 });
                const boneWhite = new THREE.MeshStandardMaterial({ color: 0xffffff });

                for (let i = 0; i < 4; i++) {
                    const coil = new THREE.Mesh(new THREE.TorusGeometry(1.4 - i*0.25, 0.3, 8, 32), snakeGreen);
                    coil.rotation.x = Math.PI / 2; coil.position.y = i * 0.25;
                    this.targetGroup.add(coil);
                }
                const neck = new THREE.Mesh(new THREE.CylinderGeometry(0.3, 0.6, 3.5, 12), snakeGreen);
                neck.position.y = 2.0; this.targetGroup.add(neck);
                const hoodGroup = new THREE.Group();
                const hoodFlare = new THREE.Mesh(new THREE.SphereGeometry(1.1, 16, 16), snakeGreen);
                hoodFlare.scale.set(1.1, 1.6, 0.25); hoodGroup.add(hoodFlare);
                const hoodMark = new THREE.Mesh(new THREE.TorusGeometry(0.4, 0.05, 8, 32), lightGreen);
                hoodMark.position.z = 0.26; hoodGroup.add(hoodMark);
                hoodGroup.position.set(0, 3.8, 0.1); this.targetGroup.add(hoodGroup);
                const head = new THREE.Mesh(new THREE.BoxGeometry(0.8, 0.5, 1.1), snakeGreen);
                head.position.set(0, 4.0, 0.5); head.name = "hitbox"; this.targetGroup.add(head);
                const fangL = new THREE.Mesh(new THREE.ConeGeometry(0.08, 0.3, 8), boneWhite);
                fangL.position.set(-0.25, 3.75, 0.9); fangL.rotation.x = Math.PI; this.targetGroup.add(fangL);
                const fangR = fangL.clone(); fangR.position.x = 0.25; this.targetGroup.add(fangR);
                this.tongue = new THREE.Group();
                const tBase = new THREE.Mesh(new THREE.BoxGeometry(0.05, 0.02, 0.6), new THREE.MeshBasicMaterial({ color: 0xff0033 }));
                this.tongue.add(tBase);
                const tForkL = tBase.clone(); tForkL.scale.set(1,1,0.5); tForkL.position.set(-0.1, 0, 0.4); tForkL.rotation.y = 0.5;
                this.tongue.add(tForkL);
                const tForkR = tBase.clone(); tForkR.scale.set(1,1,0.5); tForkR.position.set(0.1, 0, 0.4); tForkR.rotation.y = -0.5;
                this.tongue.add(tForkR);
                this.tongue.position.set(0, 3.85, 1.1); this.targetGroup.add(this.tongue);
                const leftEye = new THREE.Mesh(new THREE.SphereGeometry(0.1), glowingRed);
                leftEye.position.set(-0.25, 4.15, 1.0); this.targetGroup.add(leftEye);
                const rightEye = new THREE.Mesh(new THREE.SphereGeometry(0.1), glowingRed);
                rightEye.position.set(0.25, 4.15, 1.0); this.targetGroup.add(rightEye);

                this.scene.add(this.targetGroup);
                this.targetGroup.position.set(c.wX, 0, c.wZ);
                const playerPos = Game.getCoords(Game.state.position);
                const angle = Math.atan2(playerPos.wX - c.wX, playerPos.wZ - c.wZ);
                this.targetGroup.rotation.set(0, angle, 0); 
                
                this.targetGroup.updateMatrixWorld(true);
                SoundEngine.playHiss();
                if (initial) this.camera.lookAt(c.wX, 4.0, c.wZ);
            },

            startCombat(head, tail) {
                this.combat.active = true; 
                this.combat.enemyHp = Math.floor(Math.abs(head - tail) / 5) + 2; 
                this.buildBoard();
                const pC = Game.getCoords(Game.state.position);
                this.camera.position.set(pC.wX, 1.8, pC.wZ);
                this.camera.rotation.set(0,0,0);
                this.relocateTarget(true);
                document.getElementById('board-layer').style.opacity = 0;
                document.getElementById('board-hud').style.opacity = 0;
                document.getElementById('fps-layer').style.opacity = 1;
                document.getElementById('fps-layer').style.pointerEvents = 'auto';
                document.getElementById('fps-hud').style.display = 'block';
                document.getElementById('enemy-hp').innerText = this.combat.enemyHp;
            },

            relocateTarget(initial = false) {
                let r; do { r = Math.floor(Math.random() * 100) + 1; } while (r === Game.state.position);
                this.spawnTarget(r, initial);
                gsap.fromTo(this.targetGroup.scale, { x: 0, y: 0, z: 0 }, { x: 1, y: 1, z: 1, duration: 0.4, ease: "back.out(1.7)" });
            },

            handleFire() {
                if (!this.combat.active || Game.state.bullets <= 0) { SoundEngine.playClick(); return; }
                Game.state.bullets--; Game.updateUI(); SoundEngine.playShot();
                gsap.fromTo(this.gunGroup.position, { z: -0.5 }, { z: -0.3, duration: 0.05, yoyo: true, repeat: 1 });
                const ray = new THREE.Raycaster();
                ray.setFromCamera(new THREE.Vector2(0, 0), this.camera);
                const hits = ray.intersectObjects(this.targetGroup.children, true);
                if (hits.length > 0) {
                    this.combat.enemyHp--; SoundEngine.playHit();
                    document.getElementById('enemy-hp').innerText = this.combat.enemyHp;
                    const hitbox = this.targetGroup.getObjectByName("hitbox");
                    if (hitbox) {
                        const original = hitbox.material.color.clone();
                        hitbox.material.color.set(0xffffff);
                        setTimeout(() => { if(hitbox && hitbox.material) hitbox.material.color.copy(original); }, 100);
                    }
                    if (this.combat.enemyHp <= 0) setTimeout(() => this.endCombat(true), 300);
                    else setTimeout(() => this.relocateTarget(), 150);
                } else {
                    Game.state.hp--; Game.updateUI(); SoundEngine.playHurt();
                    gsap.fromTo("#damage-overlay", { opacity: 0.7 }, { opacity: 0, duration: 0.5 });
                    if (Game.state.hp <= 0) this.endCombat(false, "The guardian's fangs have sealed your doom.");
                }
                if (Game.state.bullets === 0 && this.combat.enemyHp > 0 && this.combat.active) {
                    setTimeout(() => { if(this.combat.active) this.endCombat(false, "Out of ammunition! The temple expels you."); }, 600);
                }
            },

            endCombat(win, msg = "") {
                this.combat.active = false; 
                if(document.pointerLockElement) document.exitPointerLock();
                document.getElementById('fps-layer').style.opacity = 0;
                document.getElementById('fps-hud').style.display = 'none';
                document.getElementById('board-layer').style.opacity = 1;
                document.getElementById('board-hud').style.opacity = 1;
                if (win) {
                    Game.state.hp = Game.state.maxHp; Game.updateUI();
                    Game.setStatus("Guardian Overcome! Advancing...");
                    setTimeout(() => Game.movePlayer(6), 800);
                } else {
                    document.getElementById('death-reason').innerText = msg;
                    document.getElementById('death-modal').style.display = 'block';
                }
            },

            animate() {
                requestAnimationFrame(() => this.animate());
                const time = Date.now() * 0.003;
                if (this.renderer && this.scene && this.camera) {
                    // INFESTATION ANIMATION
                    this.crawlingSnakes.forEach(s => {
                        const crawlTime = time + s.offset;
                        s.segments.forEach((seg, i) => {
                            seg.position.x = Math.sin(crawlTime * 2 + i * 0.5) * 0.2;
                        });
                        s.group.position.x += Math.sin(crawlTime * 0.2) * 0.01;
                        s.group.rotation.y += Math.cos(crawlTime * 0.1) * 0.002;
                    });

                    if (this.targetGroup && this.combat.active) {
                        this.targetGroup.position.y = Math.sin(time) * 0.15;
                        if (this.tongue) {
                            const tVal = Math.sin(time * 5);
                            this.tongue.scale.z = tVal > 0.5 ? 1.5 : 0.1;
                            this.tongue.position.z = 1.1 + (this.tongue.scale.z * 0.1);
                        }
                    }
                    this.renderer.render(this.scene, this.camera);
                }
            }
        };

        window.onload = () => Game.init();
    </script>
